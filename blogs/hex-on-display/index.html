D:    0101 1110                 0x5e  
C:    0011 1001                 0x39  
B:    0111 1100                 0x7c  
A:    0111 0111                 0x77  
9:    0110 1111                 0x6f  
8:    0111 1111                 0x7f  
7:    0000 0111                 0x07  
6:    0111 1101                 0x7d  
5:    0110 1101                 0x6d  
4:    0110 0110                 0x66  
3:    0100 1111                 0x4f  
2:    0101 1011                 0x5b  
1:    0000 0110                 0x06  
0:    0011 1111                 0x3f
